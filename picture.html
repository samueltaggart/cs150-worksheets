/Users/steve/Downloads/picture.py

Create and display simple pictures. Supported color names can be found
at the bottom of
https://pillow.readthedocs.io/en/stable/_modules/PIL/ImageColor.html
 
Example usage:
from picture import *
 
new_picture(400, 300)  # Create a blank, 800x600 picture
draw_text(200, 20, "A fun little picture!", 16)
set_fill_color("burlywood")
set_outline_color("dark red")
set_pen_width(8)
 
fill_rectangle(10, 30, 100, 200)
set_outline_color("green")
draw_square(20, 40, 100)
display()
delay(2000)  # Wait for 2000 ms = 2 seconds
 
set_position(100, 150)
set_outline_color("purple")
set_direction(0)
draw_forward(100)
rotate(30)
draw_forward(75)
rotate(30)
draw_forward(50)
rotate(30)
draw_forward(25)
rotate(30)
draw_forward(12)
 
set_pen_width(2)
set_outline_color("aquamarine")
draw_polygon([(200, 100), (250, 100), (225, 150)])
 
save_picture("example.png")
display()

blank_image(width, height)
Returns a blank image with the given width and height.
Example:
    image = blank_image(200, 300)
    for y in range (image_height(image)):
        for x in range(image_width(image)):
            set_pixel(image, x, y, (255, 0, 0))
    draw_image(100, 100, image)
change_picture_size(width, height)
Change the size of the picture to be width and height.

Example:
    delay(500) # Half a second
display()
Draw the current picture.
Example:
    display()
draw_circle(x, y, r)
Draw a circle centered at (x, y) with radius r in the current outline
color.
Example:
    draw_circle(100, 200, 10)
draw_forward(distance)
Draw a line starting from the current pen position in the current direction
the given distance.
Example:
    set_direction(0)
    set_position(100, 100)
    draw_forward(10)
    rotate(90)
    draw_forward(10)
    rotate(90)
    draw_forward(10)
draw_image(x, y, image)
Draws the image at (x, y).
Example:
    image = load_image("pretty_image.png")
    width = image_width(image)
    height = image_height(image)
    new_picture(width, height)
    draw_image(0, 0, image)
draw_line(x1, y1, x2, y2)
Draws a line from (x1, y1) to (x2, y2) using the current outline color and
pen width.
Example:
    draw_line(10, 20, 100, 100)
draw_oval(x, y, hrad, vrad)
Draw an oval centered at (x, y) with horizantal radius hrad and vertical
radius vrad
in the current outline color.
Example:
    draw_oval(100, 200, 10, 20)
draw_polygon(vertices)
Draw a polygon defined by a list of vertices in the current outline color.
Example:
    draw_polygon([(10, 10), (20, 10), (15, 20)])
draw_rectangle(x, y, w, h)
Draw a rectangle with corners (x, y), (x+w, y), (x, y+h), and (x+w, y+h) in
the current outline color.
Example:
    draw_rectangle(10, 20, 100, 50)
draw_square(x, y, side)
Draw a square with corners (x, y), (x+side, y), (x, y+side), and
(x+side, y+side) in the current outline color.
Example:
    draw_square(10, 20, 100, 50)
draw_text(x, y, text, font_size)
Draws the text at (x, y) using the given font_name and font_size in the
current outline color.
Example:

image_height(image)
Returns the number of pixels in each column of the image.
Example:
    image = load_image("pretty_image.png")
    width = image_width(image)
    height = image_height(image)
image_width(image)
Returns the number of pixels in each row of the image.
Example:
    image = load_image("pretty_image.png")
    width = image_width(image)
    height = image_height(image)
load_image(path)
Create an image by loading it from the file system at the given path.
Example:

    draw_image(100, 100, image) # Draws the image at (100, 100).
new_picture(width, height)
Create a new picture with the given width and height, clearing the existing
one.
Examples:
    new_picture(800, 600) # Creates an 800 x 600 blank picture
parse_color(*color)
Parses the various ways colors may be specified and returns a
(red, green, blue) tuple.
Examples:
    parse_color("aquamarine")    # returns (127, 255, 212)
    parse_color((245, 245, 220)) # "beige"; returns (245, 245, 220)
    parse_color(95, 158, 160)    # "CadetBlue"; returns (95, 158, 160)
    parse_color("#FF802A")       # returns (255, 128, 42)
rotate(theta)

save_image(image, path)
Save an image by writing it to the file system.
WARNING: This _will_ overwrite existing files so be CAREFUL!
Example:
    image = blank_image(200, 200)
    for idx in range(200):
        set_pixel(image, idx, idx, (25, 183, 200))
    save_image("fancy.png")

set_fill_color(*color)
Set the fill color. The arguments are one of a string naming the color,
a (red, green, blue) tuple, or three arguments specifying red, green, and
blue.
Examples:

    set_fill_color(95, 158, 160) # "CadetBlue"
